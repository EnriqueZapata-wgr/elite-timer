## 4. Tareas quirúrgicas

### 4.1 Crear helper `yo-image-picker.ts`

Archivo nuevo: `src/utils/yo-image-picker.ts`

```typescript
import type { ImageSourcePropType } from 'react-native';

type Sex = 'male' | 'female' | null | undefined;
type Chronotype = 'lion' | 'bear' | 'wolf' | 'dolphin' | null | undefined;

const SEX_VARIANT_IMAGES = {
  edadAtp: {
    male: require('@/assets/images/yo/edad-atp-el.png'),
    female: require('@/assets/images/yo/edad-atp-ella.png'),
  },
  composicion: {
    male: require('@/assets/images/yo/composicion-el.png'),
    female: require('@/assets/images/yo/composicion-ella.png'),
  },
};

const CHRONOTYPE_IMAGES = {
  lion: require('@/assets/images/yo/cronotipo-leon.png'),
  bear: require('@/assets/images/yo/cronotipo-oso.png'),
  wolf: require('@/assets/images/yo/cronotipo-lobo.png'),
  dolphin: require('@/assets/images/yo/cronotipo-delfin.png'),
};

export function pickEdadAtpImage(sex: Sex): ImageSourcePropType {
  return SEX_VARIANT_IMAGES.edadAtp[sex ?? 'male'];
}

export function pickComposicionImage(sex: Sex): ImageSourcePropType {
  return SEX_VARIANT_IMAGES.composicion[sex ?? 'male'];
}

export function pickChronotypeImage(chronotype: Chronotype): ImageSourcePropType {
  return CHRONOTYPE_IMAGES[chronotype ?? 'lion'];
}
```

Default `male` y `lion` si datos faltan (no romper).
