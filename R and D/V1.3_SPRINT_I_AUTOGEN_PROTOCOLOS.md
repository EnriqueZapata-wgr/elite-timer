#### A.1 Detección + resolución de conflictos de horario

Después del ajuste de cronotipo y orden, detectar múltiples eventos con misma hora `HH:MM` y separarlos con +5min. Prioridad de qué se mueve:

```typescript
// Prioridad (queda en su hora, los otros se mueven):
// action > meal > exercise > supplement > hydration > sunlight > reminder > restriction

function resolveTimeConflicts(actions: PlanAction[]): void {
  const byTime = new Map<string, PlanAction[]>();
  for (const a of actions) {
    const t = a.scheduled_time;
    if (!byTime.has(t)) byTime.set(t, []);
    byTime.get(t)!.push(a);
  }
  for (const [time, group] of byTime) {
    if (group.length <= 1) continue;
    // Ordenar por prioridad (mayor prioridad se queda en la hora original)
    group.sort((a, b) => actionPriority(b) - actionPriority(a));
    let offset = 5;
    for (let i = 1; i < group.length; i++) {
      group[i].scheduled_time = addMinutesToTime(time, offset);
      offset += 5;
    }
  }
}
```

#### A.4 Generar 2ª y 3ª comida si hay feeding window

Si el template solo trae desayuno pero hay feeding window de 8h, insertar comida del mediodía + cena automáticamente:

```typescript
function ensureMeals(actions: PlanAction[], feedingWindow: {start: string; end: string}): PlanAction[] {
  const meals = actions.filter(a => classifyAction(a.name, a.category, a.scheduled_time) === 'meal');
  if (meals.length >= 3) return actions; // ya hay suficientes
  
  const startMin = timeToMinutes(feedingWindow.start);
  const endMin = timeToMinutes(feedingWindow.end);
  const windowLen = endMin - startMin;
  
  const desiredMeals: {name: string; timeMin: number}[] = [];
  if (meals.length === 0) {
    // Ninguna comida → 3 espaciadas
    desiredMeals.push({name: 'Romper ayuno — comida limpia', timeMin: startMin});
    desiredMeals.push({name: 'Comida principal', timeMin: startMin + Math.floor(windowLen * 0.5)});
    desiredMeals.push({name: 'Cena ligera', timeMin: endMin - 60});
  } else if (meals.length === 1) {
    // Solo desayuno → agregar mediodía + cena
    const firstMealMin = timeToMinutes(meals[0].scheduled_time);
    desiredMeals.push({name: 'Comida principal', timeMin: firstMealMin + Math.max(240, Math.floor(windowLen * 0.4))});
    desiredMeals.push({name: 'Cena ligera', timeMin: endMin - 60});
  } else if (meals.length === 2) {
    // Falta una intermedia
    const middle = Math.floor((timeToMinutes(meals[0].scheduled_time) + timeToMinutes(meals[1].scheduled_time)) / 2);
    desiredMeals.push({name: 'Snack proteico', timeMin: middle});
  }
  
  const newMeals: PlanAction[] = desiredMeals.map(m => ({
    id: generateUUID(),
    name: m.name,
    category: 'nutrition',
    scheduled_time: minutesToTime(m.timeMin),
    duration_min: 15,
    instructions: 'Proteína + grasas + vegetales. Ajustar cantidad al hambre real.',
    link_type: 'food_scan',
    protocol_name: 'Auto (feeding window)',
    completed: false, completed_at: null, skipped: false,
    order: 50,
    notify_minutes_before: 15,
  }));
  
  return [...actions, ...newMeals];
}
```

#### A.5 Rellenar huecos >4h con recordatorios livianos

Detectar gaps mayores a 4h en horario diurno (9am–20pm) e insertar recordatorios opcionales:

```typescript
function fillDaytimeGaps(actions: PlanAction[]): PlanAction[] {
  const dayActions = actions
    .filter(a => {
      const t = timeToMinutes(a.scheduled_time);
      return t >= 9 * 60 && t <= 20 * 60;
    })
    .sort((a, b) => timeToMinutes(a.scheduled_time) - timeToMinutes(b.scheduled_time));
  
  const fillers: PlanAction[] = [];
  for (let i = 0; i < dayActions.length - 1; i++) {
    const gap = timeToMinutes(dayActions[i + 1].scheduled_time) - timeToMinutes(dayActions[i].scheduled_time);
    if (gap < 240) continue; // menos de 4h, sin filler
    const midMin = timeToMinutes(dayActions[i].scheduled_time) + Math.floor(gap / 2);
    // Filler simple: hidratación (siempre útil, no invasivo)
    fillers.push({
      id: generateUUID(),
      name: 'Hidratación · vaso de agua',
      category: 'nutrition',
      scheduled_time: minutesToTime(midMin),
      duration_min: 2,
      instructions: 'Vaso de agua con pizca de sal. Mantén el metabolismo despierto.',
      link_type: null,
      protocol_name: 'Auto (gap filler)',
      completed: false, completed_at: null, skipped: false,
      order: 55,
      notify_minutes_before: 0,
    });
  }
  return [...actions, ...fillers];
}
```

**Los gap fillers son opcionales** — usuario puede eliminarlos desde /agenda si no le gustan (ya soportado por EventActionModal).

#### C.2 Extender `awardBooleanElectron` en `electron-service.ts`

Aceptar `idempotencyKey` opcional. Al hacer INSERT, ON CONFLICT DO NOTHING vía UNIQUE index.

```typescript
export async function awardBooleanElectron(
  userId: string,
  source: string,
  opts?: { idempotencyKey?: string }
): Promise<{ awarded: boolean; idempotent: boolean }> {
  const idemKey = opts?.idempotencyKey ?? generateUUID();
  const weight = ELECTRON_WEIGHTS[source];
  if (!weight) return { awarded: false, idempotent: false };
  
  const { error } = await supabase.from('electron_logs').insert({
    user_id: userId,
    date: getLocalToday(),
    source,
    electrons: weight,
    idempotency_key: idemKey,
  });
  
  // Si UNIQUE constraint violation → retry idempotente, no error
  if (error?.code === '23505' && error.message.includes('idempotency_key')) {
    return { awarded: true, idempotent: true };
  }
  if (error) throw error;
  return { awarded: true, idempotent: false };
}
```
