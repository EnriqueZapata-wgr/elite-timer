Función helper para insertar divisores:
```typescript
function insertDayPartDividers(events: AgendaEventInstance[]): Array<AgendaEventInstance | { divider: string }> {
  const out: Array<AgendaEventInstance | { divider: string }> = [];
  let lastPart: 'morning' | 'afternoon' | 'evening' | null = null;
  for (const ev of events) {
    const h = parseInt(ev.time.split(':')[0], 10);
    const part = h < 12 ? 'morning' : h < 18 ? 'afternoon' : 'evening';
    if (part !== lastPart) {
      out.push({ divider: part === 'morning' ? 'MAÑANA' : part === 'afternoon' ? 'TARDE' : 'NOCHE' });
      lastPart = part;
    }
    out.push(ev);
  }
  return out;
}
```
