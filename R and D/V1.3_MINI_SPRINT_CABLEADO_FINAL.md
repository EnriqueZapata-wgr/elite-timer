    male: require('@/assets/images/cycle/embarazo/embarazo-el.png'),
    female: require('@/assets/images/cycle/embarazo/embarazo-ella.png'),
  },
};

export function pickFitnessImage(sex: Sex): ImageSourcePropType {
  return SEX_VARIANT_IMAGES.fitness[sex ?? 'male'];
}

export function pickEmbarazoImage(sex: Sex): ImageSourcePropType {
  return SEX_VARIANT_IMAGES.embarazo[sex ?? 'female'];
}

Default para `fitness`: `male`. Default para `embarazo`: `female` (la persona embarazada es siempre biológicamente femenina; la variante `male` es para mostrar al partner masculino).
